Apply ARIMA model to perform time series analysis on COVID- India Dataser from Kaggle



import pandas as pd
import matplotlib.pyplot as plt

covid_data = pd.read_csv('covid_19_india.csv')

covid_data['Date'] = pd.to_datetime(covid_data['Date'], format='%d/%m/%y')
daily_cases = covid_data.groupby('Date')['Confirmed'].sum().reset_index()
daily_cases.set_index('Date', inplace=True)


plt.figure(figsize=(12, 6))
plt.plot(daily_cases, label='Daily Confirmed Cases')
plt.title('Daily Confirmed COVID-19 Cases in India')
plt.xlabel('Date')
plt.ylabel('Number of Cases')
plt.legend()
plt.grid(True)
plt.show()


from statsmodels.tsa.stattools import adfuller
adf_test = adfuller(daily_cases['Confirmed'])
adf_result = {
    'ADF Statistic': adf_test[0],
    'p-value': adf_test[1],
    'Critical Values': adf_test[4]
    }
print(adf_result)




from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
fig, axes = plt.subplots(1, 2, figsize=(16, 4))
plot_acf(daily_cases, ax=axes[0])
plot_pacf(daily_cases, ax=axes[1])
plt.show()




from statsmodels.tsa.arima.model import ARIMA
p, d, q = 5, 1, 0  
model = ARIMA(daily_cases, order=(p, d, q))
model_fit = model.fit()
print(model_fit.summary())


forecast = model_fit.forecast(steps=10)
print(forecast)
