Predict the number of bicycle trips across Seattle's Fremont Bridge based on weather, season, and other factors and also Figure out what we can learn about people in Seattle from hourly commute data. (The daily or hourly bicycle counts can be downloaded from http://data.seattle.gov/)





import pandas as pd


data_new = pd.read_csv("Fremont_Bridge_Bicycle_Counter.csv")


print(data_new.head())
print(data_new.columns)


data_new.columns = ["Date", "Total", "East", "West"]


data_new['Date'] = pd.to_datetime(data_new['Date'])
data_new.set_index('Date', inplace=True)


print(data_new.head())


data_new["Total"] = data_new["East"] + data_new["West"]
data_cleaned = data_new.dropna()
print(data_cleaned.describe())


import matplotlib.pyplot as plt
import seaborn as sns

sns.set()


data_cleaned.plot()
plt.ylabel("Hourly Bicycle Count")
plt.title("Hourly Bicycle Count on Fremont Bridge")
plt.show()


weekly = data_cleaned.resample("W").sum()
weekly.plot(style=[':', '--', '-'])
plt.ylabel('Weekly Bicycle Count')
plt.title("Weekly Bicycle Count on Fremont Bridge")
plt.show()
